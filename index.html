// ========================================== // CONFIGURACIÓN DE TU BACKEND (CLOUD RUN) // ========================================== const API_BASE_URL = 'https://tu-servicio-ecos-xxxxxx.a.run.app/api'; let miGraficaPrincipal; let miGraficaVelocidad; // 1. LÓGICA DE LOGIN REAL async function iniciarSesion() { const user = document.getElementById('userInput').value; const pass = document.getElementById('passInput').value; const btn = document.getElementById('btnIngresar'); document.getElementById('errorMsg').style.display = 'none'; btn.disabled = true; document.getElementById('btnTexto').innerText = "Verificando..."; document.getElementById('btnSpinner').style.display = "inline-block"; try { // Petición real a tu Cloud Run const response = await fetch(`${API_BASE_URL}/login`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: user, password: pass }) }); if (response.ok) { const data = await response.json(); // Opcional: Guardar token de sesión en localStorage // localStorage.setItem('ecos_token', data.token); document.getElementById('login').style.display = 'none'; document.getElementById('main-header').style.display = 'flex'; document.getElementById('mainContent').style.display = 'block'; document.getElementById('main-nav').style.display = 'flex'; document.getElementById('home').classList.add('active'); inicializarGraficas(); obtenerDatosDeAPI(); // Cargar Dashboard } else { throw new Error('Credenciales inválidas'); } } catch (error) { console.error("Error en login:", error); document.getElementById('errorMsg').style.display = 'block'; } finally { btn.disabled = false; document.getElementById('btnTexto').innerText = "Ingresar al Sistema"; document.getElementById('btnSpinner').style.display = "none"; } } // 2. CAMBIO DE PANTALLAS function showScreen(screenId, element) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active')); document.getElementById(screenId).classList.add('active'); if(element) element.classList.add('active'); } // 3. BUSCADOR REAL (AUTOCOMPLETADO) async function mostrarSugerencias() { const query = document.getElementById('inputBuscador').value; const lista = document.getElementById('listaResultados'); const clearBtn = document.getElementById('clearSearch'); clearBtn.style.display = query.length > 0 ? 'block' : 'none'; if (query.length < 3) { // Esperar a que escriba 3 letras para no saturar tu base de datos lista.style.display = 'none'; return; } try { const response = await fetch(`${API_BASE_URL}/sondas/buscar?q=${query}`); const resultados = await response.json(); lista.innerHTML = ''; if (resultados.length > 0) { resultados.forEach(esc => { const div = document.createElement('div'); div.className = 'search-item'; const badge = esc.status === 'ok' ? '' : ''; div.innerHTML = `
${esc.nombre}${esc.id}
${badge}`; div.onclick = () => obtenerDetalleEscuela(esc.id); lista.appendChild(div); }); lista.style.display = 'block'; } else { lista.innerHTML = '
No se encontraron sondas
'; lista.style.display = 'block'; } } catch (error) { console.error("Error buscando sondas:", error); } } // 4. OBTENER DETALLE DE UNA ESCUELA (Al seleccionarla) async function obtenerDetalleEscuela(idSonda) { document.getElementById('listaResultados').style.display = 'none'; try { const response = await fetch(`${API_BASE_URL}/sondas/${idSonda}`); const escuela = await response.json(); document.getElementById('inputBuscador').value = escuela.nombre; document.getElementById('infoEscuela').style.display = 'block'; document.getElementById('escuelaNombre').innerText = escuela.nombre; document.getElementById('escuelaID').innerText = 'ID: ' + escuela.id; const badge = document.getElementById('escuelaBadge'); if(escuela.status === 'ok') { badge.className = 'status-badge ok'; badge.innerHTML = ' Activo'; } else { badge.className = 'status-badge fail'; badge.innerHTML = ' Falla'; } const upsTxt = document.getElementById('upsStatusTxt'); const upsBg = document.getElementById('upsIconBg'); const upsCol = document.getElementById('upsIconColor'); if(escuela.ups_conectada) { upsTxt.innerText = "Conectada"; upsTxt.style.color = "#111827"; upsBg.style.background = "#1e3a5f15"; upsCol.style.color = "#1e3a5f"; } else { upsTxt.innerText = "No conectada"; upsTxt.style.color = "#b91c1c"; upsBg.style.background = "#b91c1c15"; upsCol.style.color = "#b91c1c"; } document.getElementById('velPromTxt').innerText = escuela.velocidad_promedio + " Mbps"; miGraficaVelocidad.data.datasets[0].data = escuela.historial_velocidad; miGraficaVelocidad.data.datasets[0].borderColor = escuela.status === 'ok' ? '#1e3a5f' : '#b91c1c'; miGraficaVelocidad.update(); } catch (error) { console.error("Error obteniendo detalle:", error); } } function limpiarBusqueda() { document.getElementById('inputBuscador').value = ''; document.getElementById('listaResultados').style.display = 'none'; document.getElementById('clearSearch').style.display = 'none'; document.getElementById('infoEscuela').style.display = 'none'; document.getElementById('inputBuscador').focus(); } document.addEventListener('click', function(event) { const searchBox = document.querySelector('.search-box'); if (!searchBox.contains(event.target)) { document.getElementById('listaResultados').style.display = 'none'; } }); // 5. INICIALIZAR GRÁFICAS function inicializarGraficas() { if(!miGraficaPrincipal) { const ctxMain = document.getElementById('mainChart').getContext('2d'); miGraficaPrincipal = new Chart(ctxMain, { type: 'doughnut', data: { labels: ['Activas', 'Caídas'], datasets: [{ data: [0, 0], backgroundColor: ['#16a34a', '#b91c1c'], borderWidth: 0, cutout: '82%' }] }, options: { plugins: { legend: { display: false } }, maintainAspectRatio: false } }); } if(!miGraficaVelocidad) { const ctxSpeed = document.getElementById('speedChart').getContext('2d'); miGraficaVelocidad = new Chart(ctxSpeed, { type: 'line', data: { labels: ['-5h', '-4h', '-3h', '-2h', '-1h', 'Ahora'], datasets: [{ label: 'Mbps', data: [], borderColor: '#1e3a5f', borderWidth: 2, tension: 0.1, pointRadius: 0 }] }, options: { plugins: { legend: { display: false } }, scales: { y: { display: true, position: 'right', border:{display:false}, grid:{color:'#f1f5f9'} }, x: { grid: { display: false }, border:{display:false} } }, maintainAspectRatio: false } }); } } // 6. OBTENER DATOS DEL DASHBOARD PRINCIPAL async function obtenerDatosDeAPI() { try { const response = await fetch(`${API_BASE_URL}/resumen`); const data = await response.json(); document.getElementById('dashConectadas').textContent = data.sondas_activas.toLocaleString(); document.getElementById('dashCaidas').textContent = data.sondas_caidas.toLocaleString(); document.getElementById('dashTotal').textContent = data.total.toLocaleString(); miGraficaPrincipal.data.datasets[0].data = [data.sondas_activas, data.sondas_caidas]; miGraficaPrincipal.update(); } catch (error) { console.error("Error obteniendo dashboard:", error); } }
